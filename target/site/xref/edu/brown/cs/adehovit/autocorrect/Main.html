60    private static FreeMarkerEngine createEngine() {
61      Configuration config = new Configuration();
62      File templates = new File("src/main/resources/spark/template/freemarker");
63      try {
64        config.setDirectoryForTemplateLoading(templates);
65      } catch (IOException ioe) {
66        System.out.printf(
67                "ERROR: Unable use %s for template loading.\n", templates);
68        System.exit(1);
69      }
70      return new FreeMarkerEngine(config);

73    /**
74     * Prepares for autocorrect by reading options and establishing a trie.
75     */
76    private void run() {
77      OptionParser parser = new OptionParser();
78  
79      //parser.accepts("led").withRequiredArg().ofType( Integer.class );
80      led = parser.accepts("led").withRequiredArg().ofType(Integer.class);
81      parser.accepts("prefix");
82      parser.accepts("whitespace");
83      parser.accepts("smart");
84      parser.accepts("gui");
85      OptionSpec<File> fileSpec = parser.nonOptions().ofType(File.class);
86      options = parser.parse(args);
87  
88      List<File> dicts = options.valuesOf(fileSpec);
89      if (dicts.size() < 1) {
90        System.out.println("ERROR: Please specify a file");
91        System.exit(1);
92      }
93      if (options.has("led")) {
94        if (led.value(options) < 0) {
95          System.out.println("ERROR: levenshtein distance less than 0");
96          System.exit(1);
97        }
98      }
99  
100     Reader red = new Reader();
101     List<List<String>> corpuses = new ArrayList<List<String>>();
102     List<String> words = new ArrayList<String>();
103     for (File dict : dicts) {
104       try {
105         words = red.read(dict);
106       } catch (FileNotFoundException e) {
107         System.out.println("ERROR: Corpus " + dict + " not found");
108         System.exit(1);
109       } catch (IOException e) {
110         System.out.println("ERROR: file error encountered");
111         System.exit(1);
112       }
113       corpuses.add(words);
114     }
115     main = new WordData(corpuses);
116 
117     if (options.has("gui")) {
118       runSparkServer();
119     } else {
120       try {
121         InputStreamReader isr = new InputStreamReader(System.in, "UTF-8");
122         BufferedReader br = new BufferedReader(isr);
123         String line = "";
124         System.out.println("Ready");
125         while ((line = br.readLine()) != null && line.length() > 0) {
126           List<String> answers = process(line);
127           List<String> parts = new ArrayList<String>();
128           Reader.parseLine(parts, line);
129 
130           for (int i = 0; i < answers.size(); i++) {
131 
132             StringBuffer buf = new StringBuffer();
133             for (int c = 0; c < parts.size() - 1; c++) {
134               buf.append(parts.get(c));
135               buf.append(" ");
136             }
137             String result = buf.toString() + answers.get(i);
138             System.out.println(result);
139           }
140           System.out.println();
141         }
142         isr.close();
143       } catch (IOException ioe) {
144         System.out.println("ERROR: input unreadable");
145         System.exit(1);
146       }
147     }
148   }

150   /**
151    * Takes in a line of text and autocorrects the last word.
152    *
153    * @param line the line we are looking at for corrections
154    * @return List of up to five corrections to the final word
155    */
156   public List<String> process(String line) {
157     List<String> answers = new ArrayList<String>();
158     if (!Reader.getCapsAndLetters().contains(
159             line.substring(line.length() - 1, line.length()))) {
160       return answers;
161     }
162     List<String> parts = new ArrayList<>();
163     Reader.parseLine(parts, line);
164     if (parts.size() < 1) {
165       System.out.println("ERROR: no arguements given");
166       return answers;
167     }
168     String word = parts.get(parts.size() - 1);
169     String preword = "";
170     if (parts.size() > 1) {
171       preword = parts.get(parts.size() - 2);
172     }
173     Command command = new Command(main, answers,
174             word, preword, options.has("smart"));
175     command.exact();
176     if (options.has("led")) {
177       int c = led.value(options);
178       if (c > 0) {
179         command.levenshtein(c);
180       }
181     }
182     if (options.has("prefix")) {
183       command.prefix();
184     }
185     if (options.has("whitespace")) {
186       command.whitespace();
187     }
188     return answers;
189   }

191   /**
192    * Runs the spark server.
193    */
194   private void runSparkServer() {
195     Spark.externalStaticFileLocation("src/main/resources/static");
196     Spark.exception(Exception.class, new ExceptionPrinter());
197 
198     FreeMarkerEngine freeMarker = createEngine();
199 
200     // Setup Spark Routes
201     Spark.get("/autocorrect", new FrontHandler(), freeMarker);
202     Spark.post("/result", new ResultsHandler());
203   }

205   /**

208   private static class ExceptionPrinter implements ExceptionHandler {
209     @Override
210     public void handle(Exception e, Request req, Response res) {
211       res.status(500);
212       StringWriter stacktrace = new StringWriter();
213       try (PrintWriter pw = new PrintWriter(stacktrace)) {
214         pw.println("<pre>");
215         e.printStackTrace(pw);
216         pw.println("</pre>");
217       }
218       res.body(stacktrace.toString());
219     }
220   }

234   /**
235    * Takes in text input and then returns autocorrections.
236    */
237   private class ResultsHandler implements Route {
238     @Override
239     public Object handle(final Request req, final Response res) {
240       QueryParamsMap qm = req.queryMap();
241 
242 
243       String line = qm.value("line");
244 
245       List<String> answers = new ArrayList<String>();
246       if (!line.equals("")) {
247         answers = process(line);
248       }
249       Map<String, Object> variables = new ImmutableMap.Builder()
250               .put("answers", answers).build();
251 
252       return gson.toJson(variables);
253     }
254   }
