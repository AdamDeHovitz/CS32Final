3   import java.util.ArrayList;
4   import java.util.HashMap;
5   import java.util.List;
6   import java.util.Map;
7   
8   /** WordData class stores/creates the trie as well as frequencies/bigrams.
9    *
10   * @author Adam DeHovitz
11   */
12  public class WordData {
13    private Trie trie;
14    private Map<String, Map<String, Integer>> bigram;
15    private Map<String, Integer> frequency;
16  
17    /** Constructs the worddata from a list of parsed corpuses.
18     *
19     * @param corpuses a List of Lists of words, where the words came from files.
20     */
21    public WordData(List<List<String>> corpuses) {
22      bigram = new HashMap<String, Map<String, Integer>>();
23      frequency = new HashMap<String, Integer>();
24      List<Node> nodes = new ArrayList<Node>();
25      for (List<String> corpus : corpuses) {
26        readWords(corpus, nodes);
27      }
28      trie = new Trie(nodes);
29    }
30  
31     /** Returns the bigrams.
32     *
33     * @return returns the map of bigrams
34     */
35    public Map<String, Map<String, Integer>> getBigram() {
36      return bigram;
37    }
38  
39    /** returns the frequency map.
40     *
41     * @return map that connects words to their frequency
42     */
43    public Map<String, Integer> getFrequency() {
44      return frequency;
45    }
46  
47    /** return the trie.
48     *
49     * @return the constructed trie
50     */
51    public Trie getTrie() {
52      return trie;
53    }
54  
55    /** Returns the head of the trie.
56     *
57     * @return head of trie
58     */
59    public Node getTrieHead() {
60      return trie.getHead();
61    }
62  
63    /** Takes in all the words from a corpus and updates bigram, frequency, and
64     * node list.
65     *
66     * @param words List of words to parse
67     * @param nodes List of nodes to add new nodes to
68     */
69    private void readWords(List<String> words, List<Node> nodes) {
70      for (int i = 0; i < words.size(); i++) {
71        String word = words.get(i);
72        if (i < words.size() - 1) {
73          //System.out.println("word " + i + ": " + word);
74          bigramHandler(word, words.get(i + 1));
75        }
76  
77        if (!frequency.containsKey(word)) {
78          frequency.put(word, 1);
79          nodes.add(new Node(word));
80        } else {
81          frequency.put(word, frequency.get(word) + 1);
82        }
83      }
84    }
85  
86    /**  A helper for readWords which is used to update the bigram map.
87     *
88     * @param first The word preceding
89     * @param second the word following
90     */
91    private void bigramHandler(String first, String second) {
92      Map<String, Integer> pairings = bigram.get(first);
93      if (pairings == null) {
94        pairings = new HashMap<String, Integer>();
95        pairings.put(second, 1);
96        bigram.put(first, pairings);
97      } else {
98        Integer freqOfWord = pairings.get(second);
99        if (freqOfWord == null) {
100         pairings.put(second, 1);
101       } else {
102         pairings.put(second, freqOfWord + 1);
103       }
104     }
105   }
106 }
